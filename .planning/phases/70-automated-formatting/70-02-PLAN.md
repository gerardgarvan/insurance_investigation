**Step 2: Run lintr baseline scan**
Run lintr on R/ directory, excluding R/archive/ (per D-07). The .lintr config file created in Plan 01 provides the rule configuration (object_name_linter disabled, line_length_linter(120)).

```r
Rscript -e "
library(lintr)

# Run linter on all R files (reads .lintr config automatically)
# exclusions specified in .lintr file already exclude R/archive
lint_results <- lint_dir('R', exclusions = list('R/archive'))

# Convert to data frame for analysis
lint_df <- as.data.frame(lint_results)

# Total violations
total_violations <- nrow(lint_df)

# Files affected
unique_files <- length(unique(lint_df[['filename']]))

# Unique rules triggered
unique_rules <- length(unique(lint_df[['linter']]))

# Top rules by frequency
rule_counts <- sort(table(lint_df[['linter']]), decreasing = TRUE)
top_rules <- head(rule_counts, 15)

# Print results for capture
cat('=== LINT BASELINE ===\n')
cat(sprintf('Total violations: %d\n', total_violations))
cat(sprintf('Files affected: %d\n', unique_files))
cat(sprintf('Unique rules triggered: %d\n', unique_rules))
cat('\nTop 15 rules by frequency:\n')
for (i in seq_along(top_rules)) {
  cat(sprintf('  %d. %s: %d\n', i, names(top_rules)[i], top_rules[i]))
}

# Verify object_name_linter is NOT in results (should be disabled)
if ('object_name_linter' %in% lint_df[['linter']]) {
  cat('\nWARNING: object_name_linter violations found (should be disabled!)\n')
} else {
  cat('\nConfirmed: object_name_linter is disabled (no violations)\n')
}

# Check line_length_linter threshold
ll_violations <- lint_df[lint_df[['linter']] == 'line_length_linter', ]
if (nrow(ll_violations) > 0) {
  cat(sprintf('Line length violations: %d (threshold: 120 chars)\n', nrow(ll_violations)))
} else {
  cat('No line length violations (all lines under 120 chars)\n')
}
"
```
